166  std::vector<Texture> loadMaterialTextures(aiMaterial* mat, aiTextureType type, std::string typeName) {
167  std::vector<Texture> textures;
168  for (unsigned int i = 0; i < mat->GetTextureCount(type); i++) {
169  aiString str;
170  mat->GetTexture(type, i, &str);
171  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
172  bool skip = false;
173  for (unsigned int j = 0; j < textures_loaded.size(); j++) {
174  if (std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0) {
175  textures.push_back(textures_loaded[j]);
176  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
177  break;
178  }
179  }
180  if (!skip) {
181  // if texture hasn't been loaded already, load it
182  Texture texture;
183  texture.id = TextureFromFile(str.C_Str(), this->directory);
184  texture.type = typeName;

186  textures.push_back(texture);
